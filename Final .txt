    <script>
        function updateClock() {
            const systemTime = new Date();
            const thaiTimeStr = systemTime.toLocaleString("en-US", { timeZone: "Asia/Bangkok" });
            const now = new Date(thaiTimeStr);
            
            // ใช้ Millisecond เพื่อให้เข็มวินาทีเดินสมูท (เดินเรียบ)
            const ms = systemTime.getMilliseconds();
            const second = now.getSeconds() + (ms / 1000); 
            const minute = now.getMinutes();
            const hour = now.getHours();

            const secDegree = (second / 60) * 360;
            const minDegree = ((minute + second / 60) / 60) * 360;
            const hourDegree = ((hour + minute / 60) / 12) * 360;

            document.getElementById('sec').style.transform = `translateX(-50%) rotate(${secDegree}deg)`;
            document.getElementById('min').style.transform = `translateX(-50%) rotate(${minDegree}deg)`;
            document.getElementById('hour').style.transform = `translateX(-50%) rotate(${hourDegree}deg)`;

            const displaySec = now.getSeconds();
            document.getElementById('digital').innerText = 
                String(hour).padStart(2, '0') + ':' + 
                String(minute).padStart(2, '0') + ':' + 
                String(displaySec).padStart(2, '0');

            const options = { 
                timeZone: 'Asia/Bangkok',
                weekday: 'long', 
                year: 'numeric', 
                month: 'long', 
                day: 'numeric' 
            };
            document.getElementById('date').innerText = systemTime.toLocaleDateString('th-TH', options);
        }

        setInterval(updateClock, 50); // อัปเดตบ่อยๆ ให้เข็มวินาทีเดินเรียบ
        updateClock(); 
    </script>
